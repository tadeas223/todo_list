project
----
id PK int
name varchar(50)
locked bool

board
----
id PK int
name varchar(50)
project_id FK >- project

task
----
id PK int
name varchar(50)
desc varchar(512)
# enum(todo, doing, done, backlog)
state enum
progress decimal(2,1)
finish_date NULL datetime
board_id FK >- board

calendar
----
id PK int
name varchar(50)

calendar_task
----
id PK int
calendar_id FK >- calendar
task_id FK >- task
